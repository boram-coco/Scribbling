import numpy as np

# 구한 행렬
matrix = np.array([[0, 5, 10, 15, 20],
                   [5, 0, 5, 10, 15],
                   [10, 5, 0, 5, 10],
                   [15, 10, 5, 0, 5],
                   [20, 15, 10, 5, 0]])

# 행렬의 상관 계수 행렬 구하기
corr_matrix = np.corrcoef(matrix)

# 각 행렬의 값을 corr로 나누기
result = matrix / corr_matrix
result = np.exp(-result)
# 결과 확인
print(result)
